# 供应商管理看板浏览器演示 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个无需安装依赖、双击 `index.html` 即可运行的供应商管理看板演示，支持管理决策版与业务运营版切换、组件内筛选、下钻反馈和自洽 Mock 数据。

**Architecture:** 使用经典 `<script>` 顺序加载的静态单页应用，所有模块挂载到 `window.SupplierDashboard`，从而兼容 `file://` 直接打开。业务数据、统计选择器、视图状态、SVG 图表和页面组件分层组织；管理版与运营版共享数据和基础组件，但使用不同视图编排。测试使用 Node 内置 `node:test` 和 `vm` 加载同一套浏览器脚本，不引入第三方运行时或测试依赖。

**Tech Stack:** HTML5、CSS3、原生 JavaScript、内联 SVG、Node.js 20+ 内置测试运行器、PowerShell 静态资源校验脚本。

## Global Constraints

- 页面必须可以通过双击 `index.html` 直接打开，不要求本地服务器。
- 不使用 CDN、在线字体、外部图片、第三方图表库或任何需要联网的资源。
- 不使用 `type="module"`、`fetch()` 读取本地文件或其他会被 `file://` 安全策略阻止的能力。
- 演示不复刻系统蓝色顶栏；页面仅保留标题、当前数据范围说明和“管理视角 / 我的工作台”切换。
- 页面不设置全局筛选栏；筛选条件只允许出现在对应组件标题栏。
- 管理决策版统计当前用户权限组织内全部供应商。
- 业务运营版只统计当前人员负责的供应商。
- 注册完成供应商数必须来自供应商“注册状态”等于“注册完成”。
- 不展示绩效达标率。
- 绩效任务进度允许汇总全部采购品类。
- 绩效得分、等级、排名和 KPI 分析必须强制单选采购品类。
- 风险预警沿用履约项目管理首页的风险等级、LS 评价和近期风险列表结构。
- 所有统计卡、图表和明细必须由同一份 Mock 数据计算，不允许组件各自维护独立数字。
- 主要演示宽度为 1366–1920px；1024–1365px 保持可用；小于 1024px 允许组件纵向堆叠。
- 当前工作区不是 Git 仓库；执行 Task 1 时初始化 Git，后续每个任务完成后提交。

```js
const fs = require("node:fs");
const path = require("node:path");
const vm = require("node:vm");

const root = path.resolve(__dirname, "../..");

function loadApp(relativeFiles) {
  const document = {
    getElementById() {
      return null;
    }
  };
  const window = { document };
  window.window = window;

  const context = vm.createContext({
    window,
    document,
    console,
    Date,
    Intl,
    Math,
    structuredClone,
    setTimeout,
    clearTimeout
  });

  for (const relativeFile of relativeFiles) {
    const filename = path.join(root, relativeFile);
    const source = fs.readFileSync(filename, "utf8");
    vm.runInContext(source, context, { filename });
  }

  return { window, document };
}

module.exports = { loadApp };
```

Create `scripts/namespace.js`:

```js
(function bootstrapSupplierDashboard(global) {
  "use strict";

  const namespace = global.SupplierDashboard || {};

  namespace.util = {
    escapeHtml(value) {
      return String(value ?? "")
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    },
    formatInteger(value) {
      return new Intl.NumberFormat("zh-CN", {
        maximumFractionDigits: 0
      }).format(Number(value || 0));
    },
    formatPercent(value) {
      return `${(Number(value || 0) * 100).toFixed(1)}%`;
    }
  };

  global.SupplierDashboard = namespace;
})(window);
```

```js
  const data = window.SupplierDashboard.data;
  const supplierIds = new Set(data.suppliers.map((item) => item.id));
  const categoryIds = new Set(data.categories.map((item) => item.id));

  assert.equal(data.suppliers.length, 12);

    assert.equal(supplierIds.has(risk.supplierId), true);
    assert.equal(risk.lsScore, risk.likelihood * risk.severity);
```

```js
(function defineMetrics(global) {
  "use strict";

  const ns = global.SupplierDashboard;

  function uniqueSupplierCount(records) {
    return new Set(records.map((item) => item.supplierId)).size;
  }

  function getSupplierScope(data, view) {
    if (view === "operations") {
      return data.suppliers.filter((item) => item.ownerId === data.currentUserId);
    }
    return data.suppliers.filter((item) => data.managementOrgIds.includes(item.orgId));
  }

  function recordsForSuppliers(records, suppliers) {
    const ids = new Set(suppliers.map((item) => item.id));
    return records.filter((item) => ids.has(item.supplierId));
  }

  function getPerformanceTaskSummary(data, suppliers, categoryId) {
    let tasks = recordsForSuppliers(data.performanceTasks, suppliers);
    if (categoryId && categoryId !== "all") {
      tasks = tasks.filter((item) => item.categoryId === categoryId);
    }
    const completed = tasks.filter((item) => item.status === "completed").length;
    return {
      total: tasks.length,
      completed,
      inProgress: tasks.filter((item) => item.status === "in_progress").length,
      overdue: tasks.filter((item) => item.status === "overdue").length,
      periodic: tasks.filter((item) => item.type === "periodic").length,
      oneOff: tasks.filter((item) => item.type === "one-off").length,
      completionRate: tasks.length ? completed / tasks.length : 0
    };
  }

  function getSummary(data, suppliers) {
    const risks = recordsForSuppliers(data.risks, suppliers).filter(
      (item) => item.status === "open"
    );
    const remediations = recordsForSuppliers(data.remediations, suppliers).filter(
      (item) => item.status !== "completed"
    );
    const tasks = getPerformanceTaskSummary(data, suppliers, "all");

    return {
      total: suppliers.length,
      registrationCompleted: suppliers.filter(
        (item) => item.registrationStatus === "注册完成"
      ).length,
      riskSuppliers: uniqueSupplierCount(risks),
      blacklisted: suppliers.filter((item) => item.blacklisted).length,
      remediationSuppliers: uniqueSupplierCount(remediations),
      performanceCompletion: tasks.completionRate,
      pendingWorkflows: recordsForSuppliers(data.workflows, suppliers).filter(
        (item) => item.status !== "已完成"
      ).length,
      overdueWorkflows: recordsForSuppliers(data.workflows, suppliers).filter(
        (item) => item.status === "已逾期"
      ).length,
      expiringCertificates: suppliers.filter(
        (item) => item.certificateExpiry >= data.today &&
          item.certificateExpiry <= "2026-07-25"
      ).length
    };
  }

  function getGrade(score, grades) {
    return grades.find((grade) => score >= grade.min) || grades.at(-1);
  }

  function groupCount(items, key) {
    return items.reduce((accumulator, item) => {
      const value = item[key];
      accumulator[value] = (accumulator[value] || 0) + 1;
      return accumulator;
    }, {});
  }

  ns.metrics = {
    getSupplierScope,
    recordsForSuppliers,
    getPerformanceTaskSummary,
    getSummary,
    getGrade,
    groupCount
  };
})(window);
```

```js
(function defineSelectors(global) {
  "use strict";

  const ns = global.SupplierDashboard;
  const { metrics } = ns;

  function supplierMap(suppliers) {
    return Object.fromEntries(suppliers.map((item) => [item.id, item]));
  }

  function performanceByCategory(data, suppliers, categoryId) {
    if (!categoryId || categoryId === "all") {
      throw new Error("绩效得分、等级、排名和KPI分析的采购品类必须单选");
    }

    const category = data.categories.find((item) => item.id === categoryId);
    const config = data.performanceConfig[categoryId];
    const allowedIds = new Set(suppliers.map((item) => item.id));
    const assessments = data.assessments.filter(
      (item) => allowedIds.has(item.supplierId) && item.categoryId === categoryId
    );
    const suppliersById = supplierMap(suppliers);
    const latestBySupplier = {};

    for (const item of assessments) {
      const current = latestBySupplier[item.supplierId];
      if (!current || item.period > current.period) {
        latestBySupplier[item.supplierId] = item;
      }
    }

    const latest = Object.values(latestBySupplier);
    const ranking = latest
      .map((item) => ({
        supplierId: item.supplierId,
        supplierName: suppliersById[item.supplierId].name,
        score: item.score,
        grade: metrics.getGrade(item.score, config.grades)
      }))
      .sort((left, right) => right.score - left.score);

    const gradeDistribution = config.grades.map((grade) => ({
      id: grade.id,
      label: grade.label,
      color: grade.color,
      value: ranking.filter((item) => item.grade.id === grade.id).length
    }));

    const periods = [...new Set(assessments.map((item) => item.period))].sort();
    const trend = periods.map((period) => {
      const periodItems = assessments.filter((item) => item.period === period);
      const average = periodItems.reduce((sum, item) => sum + item.score, 0) /
        periodItems.length;
      return { label: period, value: Number(average.toFixed(1)) };
    });

    const kpiAverages = config.kpis.map((label, index) => {
      const scores = latest.map((item) => item.kpiScores[index]);
      return {
        label,
        value: Number(
          (scores.reduce((sum, score) => sum + score, 0) / scores.length).toFixed(1)
        )
      };
    }).sort((left, right) => left.value - right.value);

    const historyBySupplier = assessments.reduce((accumulator, item) => {
      accumulator[item.supplierId] = accumulator[item.supplierId] || [];
      accumulator[item.supplierId].push(item);
      return accumulator;
    }, {});
    const decliningSuppliers = Object.entries(historyBySupplier)
      .map(([supplierId, history]) => {
        const ordered = [...history].sort((left, right) =>
          left.period.localeCompare(right.period)
        );
        if (ordered.length < 2) return null;
        const previous = ordered.at(-2);
        const latest = ordered.at(-1);
        return {
          supplierId,
          supplierName: suppliersById[supplierId].name,
          previousScore: previous.score,
          latestScore: latest.score,
          drop: latest.score - previous.score
        };
      })
      .filter((item) => item && item.drop < 0)
      .sort((left, right) => left.drop - right.drop);

    return {
      categoryId,
      categoryName: category.name,
      gradeDistribution,
      ranking,
      trend,
      kpiAverages,
      decliningSuppliers
    };
  }

  function managementAttention(data, suppliers) {
    const supplierById = supplierMap(suppliers);
    const openRisks = metrics.recordsForSuppliers(data.risks, suppliers)
      .filter((item) => item.status === "open");
    const remediationBySupplier = Object.fromEntries(
      metrics.recordsForSuppliers(data.remediations, suppliers)
        .map((item) => [item.supplierId, item])
    );

    return suppliers
      .map((supplier) => ({
        ...supplier,
        openRiskCount: openRisks.filter((risk) => risk.supplierId === supplier.id).length,
        remediation: remediationBySupplier[supplier.id] || null,
        certificateExpired: supplier.certificateExpiry < data.today,
        certificateExpiring:
          supplier.certificateExpiry >= data.today &&
          supplier.certificateExpiry <= "2026-07-25",
        supplierName: supplierById[supplier.id].name
      }))
      .filter((item) =>
        item.openRiskCount > 0 ||
        item.remediation ||
        item.blacklisted ||
        item.certificateExpired ||
        item.certificateExpiring ||
        item.level === "淘汰" ||
        ["可剔除", "需改善"].includes(item.segment)
      )
      .sort((left, right) => {
        const leftWeight = left.openRiskCount * 10 + (left.blacklisted ? 8 : 0);
        const rightWeight = right.openRiskCount * 10 + (right.blacklisted ? 8 : 0);
        return rightWeight - leftWeight;
      });
  }

  function operationsAttention(data, suppliers) {
    return managementAttention(data, suppliers);
  }

  ns.selectors = {
    performanceByCategory,
    managementAttention,
    operationsAttention
  };
})(window);
```

```js
(function defineStore(global) {
  "use strict";

  const ns = global.SupplierDashboard;

  function createStore(initialState) {
    let state = structuredClone(initialState);
    const listeners = new Set();

    return {
      getState() {
        return state;
      },
      setState(patch) {
        state = { ...state, ...patch };
        listeners.forEach((listener) => listener(state));
      },
      updateFilter(name, value) {
        state = {
          ...state,
          filters: { ...state.filters, [name]: value }
        };
        listeners.forEach((listener) => listener(state));
      },
      subscribe(listener) {
        listeners.add(listener);
        return () => listeners.delete(listener);
      }
    };
  }

  ns.store = { createStore };
})(window);
```

```js
(function defineSvgCharts(global) {
  "use strict";

  const ns = global.SupplierDashboard;
  const esc = ns.util.escapeHtml;

  function donut(items, options = {}) {
    const total = items.reduce((sum, item) => sum + item.value, 0);
    const radius = 54;
    const circumference = 2 * Math.PI * radius;
    let offset = 0;
    const arcs = items.map((item) => {
      const length = total ? (item.value / total) * circumference : 0;
      const arc = `<circle cx="72" cy="72" r="${radius}" fill="none"
        stroke="${esc(item.color)}" stroke-width="18"
        stroke-dasharray="${length} ${circumference - length}"
        stroke-dashoffset="${-offset}" transform="rotate(-90 72 72)"/>`;
      offset += length;
      return arc;
    }).join("");

    return `<svg class="chart-svg" viewBox="0 0 320 160" role="img"
      aria-label="${esc(options.label || "分布图")}">
      <g>${arcs}</g>
      <text x="72" y="69" text-anchor="middle" class="chart-total">${total}</text>
      <text x="72" y="89" text-anchor="middle" class="chart-caption">总计</text>
      ${items.map((item, index) => `
        <g transform="translate(160 ${30 + index * 27})">
          <rect width="10" height="10" rx="2" fill="${esc(item.color)}"/>
          <text x="18" y="9">${esc(item.label)} ${item.value}</text>
        </g>`).join("")}
    </svg>`;
  }

  function bar(items, options = {}) {
    const width = 560;
    const height = 230;
    const max = Math.max(...items.map((item) => item.value), 1);
    const barWidth = Math.min(52, 420 / Math.max(items.length, 1));
    return `<svg class="chart-svg" viewBox="0 0 ${width} ${height}" role="img"
      aria-label="${esc(options.label || "柱状图")}">
      ${items.map((item, index) => {
        const x = 52 + index * (barWidth + 22);
        const barHeight = (item.value / max) * 150;
        const y = 182 - barHeight;
        return `<g data-chart-value="${esc(item.id || item.label)}">
          <rect x="${x}" y="${y}" width="${barWidth}" height="${barHeight}"
            rx="4" fill="${esc(item.color || "#247cf0")}"/>
          <text x="${x + barWidth / 2}" y="${y - 8}" text-anchor="middle">${item.value}</text>
          <text x="${x + barWidth / 2}" y="207" text-anchor="middle">${esc(item.label)}</text>
        </g>`;
      }).join("")}
    </svg>`;
  }

  function line(items, options = {}) {
    const width = 560;
    const height = 230;
    const values = items.map((item) => item.value);
    const min = Math.min(...values, 0);
    const max = Math.max(...values, 100);
    const points = items.map((item, index) => {
      const x = 45 + index * (470 / Math.max(items.length - 1, 1));
      const y = 185 - ((item.value - min) / Math.max(max - min, 1)) * 145;
      return { ...item, x, y };
    });
    return `<svg class="chart-svg" viewBox="0 0 ${width} ${height}" role="img"
      aria-label="${esc(options.label || "趋势图")}">
      <path d="${points.map((point, index) =>
        `${index ? "L" : "M"} ${point.x} ${point.y}`).join(" ")}"
        fill="none" stroke="${esc(options.color || "#247cf0")}" stroke-width="3"/>
      ${points.map((point) => `
        <circle cx="${point.x}" cy="${point.y}" r="4" fill="#fff"

        <text x="${point.x}" y="210" text-anchor="middle">${esc(point.label)}</text>
      `).join("")}
    </svg>`;
  }

          <span>${esc(cell.relation)}</span>
```

```js
(function defineRiskWidget(global) {
  "use strict";

  const ns = global.SupplierDashboard;
  const levelMeta = {
    high: { label: "较高", color: "#ff554a", tone: "danger" },
    medium: { label: "中", color: "#ffb020", tone: "warning" },
    low: { label: "低", color: "#247cf0", tone: "info" }
  };

  function riskMatrix(risks) {
    const cells = [];
    for (let severity = 5; severity >= 1; severity -= 1) {
      for (let likelihood = 1; likelihood <= 5; likelihood += 1) {
        const score = severity * likelihood;
        const count = risks.filter(
          (risk) =>
            risk.severity === severity &&
            risk.likelihood === likelihood
        ).length;
        const tone = score >= 9 ? "danger" : score >= 4 ? "warning" : "safe";
        cells.push(`
          <div class="risk-matrix-cell risk-matrix-cell--${tone}"
            title="L${likelihood} × S${severity} = ${score}">
            <small>${score}</small><strong>${count}</strong>
          </div>`);
      }
    }
    return `<div class="risk-matrix" aria-label="风险矩阵">${cells.join("")}</div>`;
  }

  ns.components.riskWidget = function riskWidget({
    data,
    suppliers,
    mode = "ls",
    filterName = "riskMode"
  }) {
    const supplierNames = Object.fromEntries(
      suppliers.map((supplier) => [supplier.id, supplier.name])
    );
    const risks = ns.metrics.recordsForSuppliers(data.risks, suppliers)
      .filter((risk) => risk.status === "open")
      .sort((left, right) => right.createdAt.localeCompare(left.createdAt));
    const distribution = ["high", "medium", "low"].map((level) => ({
      label: levelMeta[level].label,
      color: levelMeta[level].color,
      value: risks.filter((risk) => risk.level === level).length
    }));
    const controls = ns.components.selectControl({
      label: "风险评价方式",
      name: filterName,
      value: mode,
      options: [
        { value: "matrix", label: "风险矩阵" },
        { value: "ls", label: "LS评价法" }
      ]
    });
    const list = risks.length
      ? `<div class="risk-list">${risks.slice(0, 5).map((risk) => `
          <button type="button" class="risk-row" data-row-id="${risk.id}">
            ${ns.components.statusBadge(
              levelMeta[risk.level].label,
              levelMeta[risk.level].tone
            )}
            <span>
              <strong>${ns.util.escapeHtml(risk.title)}</strong>
              <small>${ns.util.escapeHtml(supplierNames[risk.supplierId])}</small>
            </span>
            <time>${ns.util.escapeHtml(risk.createdAt)}</time>
          </button>`).join("")}</div>`
      : `<div class="empty-state">暂无未关闭风险</div>`;

    return ns.components.panel({
      id: "risk-warning",
      title: "风险预警",
      controls,
      body: `
        <div class="risk-summary">
          ${mode === "matrix"
            ? riskMatrix(risks)
            : ns.charts.donut(distribution, { label: "LS风险等级分布" })}
          ${list}
        </div>`
    });
  };
```

```powershell
if ($html -match 'https?://') {

if ($html -match 'type\s*=\s*["'']module["'']') {

$references = [regex]::Matches(
  $html,

foreach ($reference in $references) {
  $resolved = Join-Path $root $reference
```
